const ERAS = ["All", "Baroque", "Classical", "Romantic", "Impressionist", "Modern"];

// group by first letter
function alphaGroups(arr, getKey) {
  const groups = {};
  [...arr].sort((a, b) => getKey(a).localeCompare(getKey(b))).forEach(item => {
    const letter = getKey(item)[0].toUpperCase();
    if (!groups[letter]) groups[letter] = [];
    groups[letter].push(item);
  });
  return groups;
}

            </nav>
          </div>

              <div className="score-ph"><div className="score-ph-icon">♩</div><div className="score-ph-label">Add page</div></div>
            </div>
          )}
          <div style={{ height: 60 }} />
        </div>
      )}
    </div>
  );
}
